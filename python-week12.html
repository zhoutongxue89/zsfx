1.3 文件指针操作

💡 概念

文件指针标记当前读取/写入的位置。使用 seek() 移动指针，tell() 获取当前位置。

文件指针操作 Python 
with open('test.txt', 'r', encoding='utf-8') as f:
    print(f"初始位置: {f.tell()}")  # 0
    
    content = f.read(5)  # 读取5个字符
    print(f"读取5字符后: {f.tell()}")  # 5
    
    f.seek(0)  # 回到开头
    print(f"seek(0)后: {f.tell()}")  # 0
    
    f.seek(0, 2)  # 移到末尾（2表示从末尾计算）
    print(f"文件末尾: {f.tell()}")
    
    # seek(offset, whence)
    # whence=0: 从开头（默认）
    # whence=1: 从当前位置
    # whence=2: 从末尾

2.2 上下文管理器原理

💡 概念

上下文管理器需要实现 __enter__ 和 __exit__ 两个方法。with语句会自动调用它们来管理资源。

自定义上下文管理器 Python 
class FileManager:
    """自定义文件管理器（实现上下文管理器协议）"""
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
        self.file = None
    
    def __enter__(self):
        """进入with块时调用"""
        print(f"打开文件: {self.filename}")
        self.file = open(self.filename, self.mode, encoding='utf-8')
        return self.file  # 返回值绑定到 as 后的变量
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        """退出with块时调用"""
        print(f"关闭文件: {self.filename}")
        if self.file:
            self.file.close()
        # 返回True可以压制异常
        return False

# 使用自定义上下文管理器
with FileManager('test.txt', 'r') as f:
    content = f.read()
    print(f"读取内容: {content[:50]}...")

# 即使发生异常，__exit__也会执行
try:
    with FileManager('test.txt', 'r') as f:
        raise ValueError("测试异常")
except ValueError as e:
    print(f"异常被捕获: {e}")

2.3 contextlib简化写法

@contextmanager装饰器 Python 
from contextlib import contextmanager

@contextmanager
def file_manager(filename, mode):
    """使用装饰器简化上下文管理器"""
    f = open(filename, mode, encoding='utf-8')
    try:
        yield f  # yield之前的代码相当于__enter__
    finally:
        f.close()  # finally块相当于__exit__

# 使用
with file_manager('test.txt', 'r') as f:
    content = f.read()
    print(content)

实战：安全的文件合并工具
文件合并 Python 
from contextlib import contextmanager

@contextmanager
def safe_file_merge(output_file, input_files):
    """安全地合并多个文件"""
    print(f"开始合并 {len(input_files)} 个文件...")
    output = open(output_file, 'w', encoding='utf-8')
    total_lines = 0
    try:
        yield output  # 提供输出文件对象给调用者
        for fname in input_files:
            with open(fname, 'r', encoding='utf-8') as inp:
                for line in inp:
                    output.write(line)
                    total_lines += 1
        print(f"合并完成: {total_lines} 行")
    finally:
        output.close()

# 使用
files_to_merge = ['file1.txt', 'file2.txt', 'file3.txt']
with safe_file_merge('merged.txt', files_to_merge) as out:
    out.write("# 合并文件头部\n")

实战：递归文件搜索器
文件搜索器 Python 
from pathlib import Path

def search_files(root_dir, extensions=None, min_size=0, max_size=None):
    """
    递归搜索文件
    
    Args:
        root_dir: 搜索根目录
        extensions: 文件扩展名列表，如 ['.py', '.txt']
        min_size: 最小文件大小（字节）
        max_size: 最大文件大小（字节）
    """
    results = []
    root = Path(root_dir)
    
    if not root.exists():
        print(f"目录不存在: {root_dir}")
        return results
    
    for path in root.rglob('*'):  # 递归遍历
        if path.is_file():
            # 检查扩展名
            if extensions and path.suffix not in extensions:
                continue
            
            # 检查大小
            size = path.stat().st_size
            if size < min_size:
                continue
            if max_size and size > max_size:
                continue
            
            results.append({
                'path': str(path),
                'name': path.name,
                'size': size,
                'modified': path.stat().st_mtime
            })
    
    return results

# 搜索示例
files = search_files(
    '.',
    extensions=['.py', '.md'],
    min_size=1000
)
print(f"找到 {len(files)} 个符合条件的文件")
for f in files[:5]:
    print(f"  {f['name']} ({f['size']/1024:.1f}KB)")

4.4 实战：CSV数据处理器

CSV数据处理 Python 
import csv
from pathlib import Path

def process_csv(input_file, output_file, transform_fn):
    """
    处理CSV文件
    
    Args:
        input_file: 输入文件路径
        output_file: 输出文件路径
        transform_fn: 转换函数，接收一行字典，返回处理后的字典或None（跳过）
    """
    with open(input_file, 'r', encoding='utf-8-sig') as fin, \
         open(output_file, 'w', newline='', encoding='utf-8-sig') as fout:
        
        reader = csv.DictReader(fin)
        writer = csv.DictWriter(fout, fieldnames=reader.fieldnames)
        writer.writeheader()
        
        count = 0
        for row in reader:
            new_row = transform_fn(row)
            if new_row:  # 返回None时跳过
                writer.writerow(new_row)
                count += 1
        
        print(f"处理完成: {count} 行")

# 使用示例：筛选成绩大于80的学生
def filter_good_students(row):
    if int(row['成绩']) > 80:
        return row  # 保留
    return None  # 跳过

process_csv('students.csv', 'good_students.csv', filter_good_students)

# 使用示例：添加平均分列
def add_average(row):
    row['等级'] = 'A' if int(row['成绩']) >= 90 else \
                  'B' if int(row['成绩']) >= 80 else \
                  'C' if int(row['成绩']) >= 60 else 'D'
    return row

process_csv('students.csv', 'with_grade.csv', add_average)

import logging

def setup_logger(name, log_file, level=logging.INFO):
    """创建logger，同时输出到控制台和文件"""
    logger = logging.getLogger(name)
    logger.setLevel(level)
    
    # 格式化器
    formatter = logging.Formatter(
        '%(asctime)s [%(levelname)s] %(name)s: %(message)s'
    )
    
    # 控制台处理器
    console = logging.StreamHandler()
    console.setFormatter(formatter)
    logger.addHandler(console)
    
    # 文件处理器
    file_handler = logging.FileHandler(log_file, encoding='utf-8')
    file_handler.setFormatter(formatter)
    logger.addHandler(file_handler)
    
    return logger

# 使用

app_logger.error('发生错误')

# 不同模块使用不同logger

实战：程序日志装饰器
日志装饰器 Python 
import logging
from functools import wraps

logger = logging.getLogger('decorator')

def log_calls(func):
    """自动记录函数调用的装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        logger.info(f"调用函数: {func.__name__}")
        logger.debug(f"  参数: args={args}, kwargs={kwargs}")
        try:
            result = func(*args, **kwargs)
            logger.info(f"  返回: {result}")
            return result
        except Exception as e:
            logger.error(f"  异常: {e}")
            raise
    return wrapper

    return a / b

文件备份工具 Python 
import os
import shutil
import hashlib
import json
import logging
from pathlib import Path
from datetime import datetime

# 配置日志
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s [%(levelname)s] %(message)s'
)
logger = logging.getLogger('backup')

class FileBackup:
    """文件备份工具"""
    
    def __init__(self, source_dir, backup_dir):
        self.source = Path(source_dir)
        self.backup = Path(backup_dir)
        self.backup.mkdir(parents=True, exist_ok=True)
        self.manifest_file = self.backup / 'manifest.json'
        self.manifest = self._load_manifest()
    
    def _load_manifest(self):
        """加载清单文件"""
        if self.manifest_file.exists():
            with open(self.manifest_file, 'r', encoding='utf-8') as f:
                return json.load(f)
        return {}
    
    def _save_manifest(self):
        """保存清单文件"""
        with open(self.manifest_file, 'w', encoding='utf-8') as f:
            json.dump(self.manifest, f, indent=2, ensure_ascii=False)
    
    def _get_file_hash(self, filepath):
        """计算文件MD5"""
        md5 = hashlib.md5()
        with open(filepath, 'rb') as f:
            for chunk in iter(lambda: f.read(8192), b''):
                md5.update(chunk)
        return md5.hexdigest()
    
    def backup_files(self, extensions=None):
        """
        备份文件
        
        Args:
            extensions: 要备份的扩展名，如 ['.py', '.md']
        """
        backed_up = 0
        skipped = 0
        
        for file_path in self.source.rglob('*'):
            if not file_path.is_file():
                continue
            
            # 检查扩展名
            if extensions and file_path.suffix not in extensions:
                continue
            
            # 计算相对路径和哈希
            rel_path = file_path.relative_to(self.source)
            file_hash = self._get_file_hash(file_path)
            
            # 检查是否需要备份（新增或修改）
            if rel_path in self.manifest:
                if self.manifest[rel_path]['hash'] == file_hash:
                    logger.debug(f"跳过(未修改): {rel_path}")
                    skipped += 1
                    continue
            
            # 复制文件
            dest_path = self.backup / rel_path
            dest_path.parent.mkdir(parents=True, exist_ok=True)
            shutil.copy2(file_path, dest_path)
            
            # 更新清单
            self.manifest[str(rel_path)] = {
                'hash': file_hash,
                'size': file_path.stat().st_size,
                'backup_time': datetime.now().isoformat(),
                'source_path': str(file_path)
            }
            
            logger.info(f"已备份: {rel_path}")
            backed_up += 1
        
        self._save_manifest()
        print(f"\n备份完成: {backed_up} 个文件已更新, {skipped} 个文件已跳过")
    
    def restore_files(self, target_dir):
        """从备份恢复文件"""
        target = Path(target_dir)
        target.mkdir(parents=True, exist_ok=True)
        
        for rel_path, info in self.manifest.items():
            src = self.backup / rel_path
            dst = target / rel_path
            
            if src.exists():
                dst.parent.mkdir(parents=True, exist_ok=True)
                shutil.copy2(src, dst)
                logger.info(f"已恢复: {rel_path}")
        
        print(f"\n恢复完成: {len(self.manifest)} 个文件")

# 使用示例
if __name__ == '__main__':
    backup = FileBackup('.', 'backup')
    backup.backup_files(extensions=['.py', '.md'])
