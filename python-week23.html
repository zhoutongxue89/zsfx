1️⃣ NumPy基础

🔑 NumPy核心

高性能多维数组库，Python数据科学的基石。

import numpy as np

# 创建数组
arr = np.array([1, 2, 3, 4, 5])
arr2d = np.array([[1, 2, 3], [4, 5, 6]])

# 数组操作
print(arr.shape)      # (5,)
print(arr2d.shape)    # (2, 3)
print(arr.sum())     # 15
print(arr.mean())    # 3.0
print(arr.max())     # 5
print(arr.min())     # 1

# 数学运算
arr + 10             # [11, 12, 13, 14, 15]
arr * 2              # [2, 4, 6, 8, 10]
arr ** 2             # [1, 4, 9, 16, 25]

# 数组切片
arr[1:3]             # [2, 3]
arr2d[0, :]          # [1, 2, 3]
arr2d[:, 1]          # [2, 5]

# 条件筛选
arr[arr > 3]         # [4, 5]
np.where(arr > 3, 'big', 'small')

7️⃣ 实战案例：销售数据分析

import pandas as pd
import matplotlib.pyplot as plt

# 读取数据
sales = pd.read_csv('sales_data.csv')

# 数据清洗
sales['date'] = pd.to_datetime(sales['date'])
sales['revenue'] = sales['price'] * sales['quantity']
sales = sales.dropna()

# 分析维度
# 1. 按月份汇总
monthly = sales.groupby(sales['date'].dt.to_period('M'))['revenue'].sum()

# 2. 按产品分类
by_category = sales.groupby('category')['revenue'].agg(['sum', 'mean', 'count'])

# 3. 按地区
by_region = sales.groupby('region')['revenue'].sum().sort_values(ascending=False)

# 4. Top 10 客户
top_customers = sales.groupby('customer')['revenue'].sum().nlargest(10)

# 可视化
fig, axes = plt.subplots(2, 2, figsize=(14, 10))

monthly.plot(ax=axes[0, 0], title='月度销售趋势')
by_category['sum'].plot(kind='bar', ax=axes[0, 1], title='分类销售')
by_region.plot(kind='pie', ax=axes[1, 0], title='地区分布', autopct='%1.1f%%')
top_customers.plot(kind='barh', ax=axes[1, 1], title='Top 10客户')

plt.tight_layout()
plt.savefig('sales_analysis.png')
plt.show()

# 输出报告
print("=== 销售分析报告 ===")
print(f"总收入: ¥{sales['revenue'].sum():,.2f}")
print(f"平均订单: ¥{sales['revenue'].mean():,.2f}")
print(f"最高单笔: ¥{sales['revenue'].max():,.2f}")
