1️⃣ 同步 vs 异步

🔑 核心概念
  • 同步（Synchronous）：按顺序执行，一步一步来，等待I/O时不干活
  • 异步（Asynchronous）：发起I/O后不等结果，继续执行其他任务，I/O完成后再处理结果

📌 生活中的类比

🍜 同步：排队买面

点完单就站在柜台等，面好了才走

等待时什么都不干（阻塞）

📱 异步：点外卖

下单后去做其他事

手机响了再去取（回调通知）

📌 代码对比

import time
import asyncio

# 同步方式：串行执行
def sync_fetch(urls):
    import requests
    results = []
    for url in urls:
        response = requests.get(url)  # 阻塞等待
        results.append(response.text)
    return results

# 异步方式：并发执行
async def async_fetch(urls):
    import aiohttp
    async with aiohttp.ClientSession() as session:
        tasks = [session.get(url) for url in urls]
        responses = await asyncio.gather(*tasks)
        return [r.text() async for r in responses]

# 时间对比
start = time.time()
# sync_fetch(['http://example.com'] * 10)  # 约10秒（每个1秒）
# async_fetch(['http://example.com'] * 10) # 约1秒（并发）
print(f"耗时: {time.time() - start:.2f}s")
✅ 异步优势
  • 单线程实现并发，资源开销小
  • 适合大量I/O等待场景
  • 不需要锁，避免锁竞争

3️⃣ async/await语法

📌 定义协程函数

import asyncio

# 协程函数：使用async def定义
async def hello():
    print("Hello")
    await asyncio.sleep(1)  # 模拟I/O
    print("World")

# 运行协程
asyncio.run(hello())
💡 重要特性
  • async def定义的函数是协程函数，调用它会返回协程对象
  • await只能在async函数内使用
  • asyncio.run()创建事件循环并运行协程

📌 await关键字

import asyncio

async def fetch_data():
    # 等待另一个协程完成
    result = await asyncio.sleep(2)  # 等待2秒
    return "数据"

async def main():
    # 顺序执行：约4秒
    # result1 = await fetch_data()
    # result2 = await fetch_data()
    
    # 并发执行：约2秒
    result1, result2 = await asyncio.gather(
        fetch_data(),
        fetch_data()
    )
    print(result1, result2)

asyncio.run(main())

📌 并发执行：gather

import asyncio
import time

async def task(n):
    print(f"任务 {n} 开始")
    await asyncio.sleep(1)
    print(f"任务 {n} 完成")
    return n * n

async def main():
    start = time.time()
    
    # 方法1：gather - 并发执行
    results = await asyncio.gather(
        task(1),
        task(2),
        task(3),
        task(4),
    )
    print(f"gather结果: {results}")  # [1, 4, 9, 16]
    
    # 方法2：create_task - 创建任务
    task1 = asyncio.create_task(task(1))
    task2 = asyncio.create_task(task(2))
    
    # 等待两个任务完成
    await task1
    await task2
    
    # 方法3：wait - 等待一组任务
    tasks = [asyncio.create_task(task(i)) for i in range(5)]
    done, pending = await asyncio.wait(tasks)
    
    print(f"总耗时: {time.time() - start:.2f}s")

asyncio.run(main())

4️⃣ 协程与事件循环

📌 创建事件循环的方式

import asyncio

# 方式1：asyncio.run (Python 3.7+)
async def main():

loop = asyncio.new_event_loop()
asyncio.set_event_loop(loop)

    loop.run_until_complete(main())
finally:
    loop.close()

# 方式3：get_event_loop

    loop = asyncio.get_running_loop()
    print(f"当前循环: {loop}")
    await asyncio.sleep(1)

asyncio.run(main())

📌 协程的嵌套调用

import asyncio

async def level3():
    print("Level 3")
    await asyncio.sleep(0.5)
    return "level3 result"

async def level2():
    print("Level 2")
    result = await level3()
    print(f"收到: {result}")
    return "level2 result"

async def level1():
    print("Level 1")
    result = await level2()

    return "level1 result"

⚠️ 常见错误

忘记使用await！协程必须被await或通过gather/create_task包装才能执行。

# 错误写法
async def main():
    hello()  # 只是创建了协程对象，没有执行

# 正确写法
async def main():
    await hello()  # 执行协程

5️⃣ Task任务管理

📌 创建Task的方式

import asyncio

async def worker(n):
    await asyncio.sleep(n)
    return f"任务{n}完成"

async def main():
    # 方式1：create_task (Python 3.7+)
    task1 = asyncio.create_task(worker(1))
    
    # 方式2：ensure_future
    task2 = asyncio.ensure_future(worker(2))
    
    # 等待任务完成
    result1 = await task1
    result2 = await task2
    
    print(result1, result2)

asyncio.run(main())

📌 Task的状态

import asyncio

async def worker():
    await asyncio.sleep(1)
    return "done"

async def main():
    task = asyncio.create_task(worker())
    
    print(f"初始状态: {task.done()}")  # False
    
    await asyncio.sleep(0.5)
    print(f"中途状态: {task.done()}")  # False
    
    await task
    print(f"最终状态: {task.done()}")  # True
    print(f"结果: {task.result()}")

asyncio.run(main())

📌 wait与wait_for

import asyncio

async def slow_task():
    await asyncio.sleep(10)
    return "完成"

async def main():
    # wait - 等待一组任务
    tasks = [asyncio.create_task(slow_task()) for _ in range(3)]
    done, pending = await asyncio.wait(tasks, timeout=2)
    
    print(f"已完成: {len(done)}")
    print(f"待完成: {len(pending)}")
    
    # 取消待完成的任务
    for task in pending:
        task.cancel()
    
    # wait_for - 超时等待
    try:
        result = await asyncio.wait_for(slow_task(), timeout=2)
    except asyncio.TimeoutError:
        print("任务超时！")

asyncio.run(main())

6️⃣ 异步上下文管理器

📌 async with

import asyncio

class AsyncResource:
    def __init__(self, name):
        self.name = name
    
    async def __aenter__(self):
        print(f"获取资源: {self.name}")
        await asyncio.sleep(0.1)  # 模拟连接
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        print(f"释放资源: {self.name}")
        await asyncio.sleep(0.1)  # 模拟关闭
        return False

async def main():
    async with AsyncResource("数据库连接") as conn:
        print(f"使用 {conn.name}")
    
    # 多个资源
    async with AsyncResource("连接1") as r1, \
               AsyncResource("连接2") as r2:
        print(f"同时使用 {r1.name} 和 {r2.name}")

asyncio.run(main())

📌 异步迭代器

import asyncio

class AsyncIterator:
    def __init__(self, items):
        self.items = items
        self.index = 0
    
    def __aiter__(self):
        return self
    
    async def __anext__(self):
        await asyncio.sleep(0.1)
        if self.index >= len(self.items):
            raise StopAsyncIteration
        item = self.items[self.index]
        self.index += 1
        return item

async def main():
    async for item in AsyncIterator([1, 2, 3, 4, 5]):
        print(f"处理: {item}")

asyncio.run(main())

# 更简单的方式：异步生成器
async def async_generator(items):
    for item in items:
        await asyncio.sleep(0.1)
        yield item

async def main2():
    async for item in async_generator([1, 2, 3]):
        print(f"生成: {item}")

asyncio.run(main2())

7️⃣ aiohttp异步HTTP

📌 基本用法

import asyncio
import aiohttp

async def fetch(session, url):
    async with session.get(url) as response:
        return await response.text()

async def main():
    urls = [
        "https://example.com",
        "https://httpbin.org/html",
        "https://httpbin.org/json",
    ]
    
    async with aiohttp.ClientSession() as session:
        # 并发请求
        tasks = [fetch(session, url) for url in urls]
        results = await asyncio.gather(*tasks)
        
        for url, content in zip(urls, results):
            print(f"{url}: {len(content)} bytes")

asyncio.run(main())

📌 带参数和错误处理

import asyncio
import aiohttp

async def fetch_with_params(session, url, params=None):
    try:
        async with session.get(url, params=params) as response:
            if response.status == 200:
                return await response.json()
            else:
                print(f"HTTP {response.status}")
                return None
    except aiohttp.ClientError as e:
        print(f"请求错误: {e}")
        return None

async def main():
    async with aiohttp.ClientSession() as session:
        # GET请求带参数
        result = await fetch_with_params(
            "https://httpbin.org/get",
            params={"key": "value", "name": "test"}
        )
        print(result)
        
        # POST请求
        async with session.post(
            "https://httpbin.org/post",
            json={"username": "test", "password": "123456"}
        ) as response:
            result = await response.json()
            print(result)

asyncio.run(main())

📌 并发限制

import asyncio
import aiohttp

async def fetch_all(urls, max_concurrent=5):
    """限制并发数量的异步请求"""
    semaphore = asyncio.Semaphore(max_concurrent)
    
    async def fetch_with_limit(session, url):
        async with semaphore:
            async with session.get(url) as response:
                return await response.text()
    
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_with_limit(session, url) for url in urls]
        return await asyncio.gather(*tasks)

async def main():
    urls = [f"https://httpbin.org/delay/{i}" for i in range(20)]
    
    # 限制同时最多5个请求
    results = await fetch_all(urls, max_concurrent=5)
    print(f"完成 {len(results)} 个请求")

asyncio.run(main())

8️⃣ 实战：异步爬虫

🔑 项目目标

使用asyncio和aiohttp构建高效异步爬虫。

import asyncio
import aiohttp
import time
from urllib.parse import urljoin, urlparse
from html.parser import HTMLParser

class LinkParser(HTMLParser):
    """提取页面中的链接"""
    def __init__(self, base_url):
        super().__init__()
        self.base_url = base_url
        self.links = set()
    
    def handle_starttag(self, tag, attrs):
        if tag == 'a':
            for attr_name, attr_value in attrs:
                if attr_name == 'href' and attr_value:
                    # 转换为绝对URL
                    full_url = urljoin(self.base_url, attr_value)
                    # 只保留同域名的链接
                    if urlparse(full_url).netloc == urlparse(self.base_url).netloc:
                        self.links.add(full_url)

class AsyncSpider:
    def __init__(self, start_url, max_depth=2, max_concurrent=10):
        self.start_url = start_url
        self.max_depth = max_depth
        self.max_concurrent = max_concurrent
        self.visited = set()
        self.session = None
        self.semaphore = asyncio.Semaphore(max_concurrent)
    
    async def fetch(self, url):
        """获取页面内容"""
        async with self.semaphore:
            try:
                async with self.session.get(url, timeout=aiohttp.ClientTimeout(total=30)) as response:
                    if response.status == 200:
                        content = await response.text()
                        return content
                    return None
            except Exception as e:
                print(f"抓取失败 {url}: {e}")
                return None
    
    async def crawl(self, url, depth=0):
        """递归爬取"""
        if depth >= self.max_depth or url in self.visited:
            return
        
        self.visited.add(url)
        print(f"[深度{depth}] {url}")
        
        content = await self.fetch(url)
        if not content:
            return
        
        # 解析链接
        parser = LinkParser(url)
        parser.feed(content)
        
        # 并发爬取子页面
        tasks = []
        for link in parser.links:
            if link not in self.visited:
                task = asyncio.create_task(self.crawl(link, depth + 1))
                tasks.append(task)
        
        if tasks:
            await asyncio.gather(*tasks)
    
    async def run(self):
        """启动爬虫"""
        async with aiohttp.ClientSession() as session:
            self.session = session
            await self.crawl(self.start_url)
        
        print(f"\n爬取完成，共 {len(self.visited)} 个页面")

if __name__ == "__main__":
    start_time = time.time()
    
    spider = AsyncSpider("https://example.com", max_depth=2, max_concurrent=5)
    asyncio.run(spider.run())
    
    print(f"总耗时: {time.time() - start_time:.2f}s")

9️⃣ asyncio最佳实践

✅ 建议
  • 使用asyncpg/aiohttp/aiomysql：异步版本的数据库和HTTP客户端
  • 限制并发数：使用Semaphore避免请求过多
  • 合理设置超时：使用asyncio.wait_for防止永久阻塞
  • 复用Session：aiohttp的ClientSession应复用
  • 避免阻塞调用：不要在async函数中直接调用同步I/O

📌 性能对比

import asyncio
import aiohttp
import time

async def benchmark():
    """对比同步和异步HTTP请求"""
    urls = ["https://httpbin.org/delay/1"] * 10
    
    # 异步方式
    start = time.time()
    async with aiohttp.ClientSession() as session:
        tasks = [session.get(url) for url in urls]
        await asyncio.gather(*tasks)
    async_time = time.time() - start
    
    print(f"异步 (10个并发): {async_time:.2f}s")
    print(f"理论串行时间: 10s")

asyncio.run(benchmark())
