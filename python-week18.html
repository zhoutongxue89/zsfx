1️⃣ 什么是装饰器

🔑 核心定义

装饰器（Decorator）是一种高级语法糖，允许在不修改原函数的情况下，给函数添加新功能。

📌 生活中的类比

就像给手机装保护壳：不需要改变手机本身，但能添加保护功能。

📌 装饰器解决的问题

# 问题：给多个函数添加计时功能
import time

def func1():
    start = time.time()
    print("执行功能1")
    time.sleep(1)
    print(f"耗时: {time.time() - start:.2f}s")

def func2():
    start = time.time()
    print("执行功能2")
    time.sleep(2)
    print(f"耗时: {time.time() - start:.2f}s")

# 问题：重复代码！每个函数都要加一遍

# 解决方案：装饰器
def timer(func):
    def wrapper():
        start = time.time()
        result = func()
        print(f"耗时: {time.time() - start:.2f}s")
        return result
    return wrapper

@timer
def func1():
    print("执行功能1")
    time.sleep(1)

@timer
def func2():
    print("执行功能2")
    time.sleep(2)

2️⃣ 函数即对象

理解装饰器前，需要理解Python的一个核心特性：函数是一等公民。

📌 函数可以赋值给变量

def greet(name):
    return f"Hello, {name}!"

# 赋值给变量
say_hello = greet

# 通过变量调用
print(say_hello("World"))  # Hello, World!

# 函数可以作为参数传递
def call_twice(func, arg):
    return func(func(arg))

print(call_twice(greet, "Python"))  # Hello, Hello, Python!

📌 函数可以嵌套定义

def outer():
    x = 10
    def inner():
        print(f"x = {x}")  # 可以访问外层变量
    inner()
    print(f"外部x = {x}")

outer()

# 闭包：内部函数记住外层变量
def outer_with_return():
    x = 10
    def inner():
        return x * 2
    return inner  # 返回函数本身

double = outer_with_return()
print(double())  # 20

📌 高阶函数

# map：对每个元素应用函数
numbers = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x ** 2, numbers))
print(squared)  # [1, 4, 9, 16, 25]

# filter：过滤元素
evens = list(filter(lambda x: x % 2 == 0, numbers))
print(evens)  # [2, 4]

# reduce：累积计算
from functools import reduce
total = reduce(lambda a, b: a + b, numbers)
print(total)  # 15

3️⃣ 简单装饰器

📌 装饰器原理

import time

# 装饰器本质：高阶函数
def my_decorator(func):
    """装饰器：接收一个函数，返回一个新函数"""
    def wrapper():
        print("=== 执行前 ===")
        result = func()  # 调用原函数
        print("=== 执行后 ===")
        return result
    return wrapper

# 使用方式1：直接包装
def say_hello():
    print("Hello!")

wrapped = my_decorator(say_hello)
wrapped()

# 使用方式2：@装饰器语法（更简洁）
@my_decorator
def say_goodbye():
    print("Goodbye!")

say_goodbye()
@my_decorator
def func():
    pass

# 等价于
def func():
    pass
func = my_decorator(func)

📌 通用装饰器

import time

def timer(func):
    """计时装饰器"""
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        print(f"{func.__name__} 耗时: {time.time() - start:.3f}s")
        return result
    return wrapper

@timer
def slow_function(n):
    time.sleep(n)
    return n * 2

result = slow_function(1)
print(f"结果: {result}")

4️⃣ 带参数的装饰器

📌 装饰器带参数：装饰器工厂

def repeat(times):
    """重复执行指定次数"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            results = []
            for _ in range(times):
                result = func(*args, **kwargs)
                results.append(result)
            return results
        return wrapper
    return decorator

@repeat(3)
def greet(name):
    return f"Hello, {name}!"

print(greet("Alice"))
# ['Hello, Alice!', 'Hello, Alice!', 'Hello, Alice!']

📌 带配置的装饰器

import time

def log(level="INFO"):
    """日志装饰器"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            print(f"[{level}] 调用 {func.__name__}")
            result = func(*args, **kwargs)
            print(f"[{level}] {func.__name__} 执行完成")
            return result
        return wrapper
    return decorator

@log("DEBUG")
def debug_function():
    print("调试信息")

@log("ERROR")
def error_function():
    print("错误信息")

debug_function()
error_function()

📌 使用类实现装饰器

class CountCalls:
    """计数装饰器"""
    def __init__(self, func):
        self.func = func
        self.calls = 0
    
    def __call__(self, *args, **kwargs):
        self.calls += 1
        print(f"{self.func.__name__} 被调用了 {self.calls} 次")
        return self.func(*args, **kwargs)

@CountCalls
def add(a, b):
    return a + b

add(1, 2)  # add 被调用了 1 次
add(3, 4)  # add 被调用了 2 次
add(5, 6)  # add 被调用了 3 次

5️⃣ 装饰器堆叠

多个装饰器可以叠加使用，执行顺序从下到上。

def deco_a(func):
    def wrapper():
        print("A: 开始")
        result = func()
        print("A: 结束")
        return result
    return wrapper

def deco_b(func):
    def wrapper():
        print("B: 开始")
        result = func()
        print("B: 结束")
        return result
    return wrapper

@deco_a
@deco_b
def my_func():
    print("my_func 执行")

my_func()
# 输出:
# A: 开始
# B: 开始
# my_func 执行
# B: 结束
# A: 结束
💡 执行顺序

装饰器从下到上执行（靠近函数先执行），但从功能上看是从外到内。

6️⃣ @functools.wraps

⚠️ 问题

装饰器会替换原函数，导致函数元信息（名字、文档、注释）丢失。

def simple_decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@simple_decorator
def my_function():
    """这是我的函数"""
    pass

print(my_function.__name__)  # wrapper （错误！）
print(my_function.__doc__)   # None （丢失！）

📌 使用wraps保留元信息

import functools

def smart_decorator(func):
    @functools.wraps(func)  # 保留原函数元信息
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@smart_decorator
def documented_function():
    """这是带文档的函数"""
    pass

print(documented_function.__name__)  # documented_function
print(documented_function.__doc__)   # 这是带文档的函数

7️⃣ 类装饰器

装饰器不仅能用于函数，也能用于类。

📌 类装饰器原理

class MyClassDecorator:
    def __init__(self, cls):
        self.cls = cls
    
    def __call__(self, *args, **kwargs):
        # 可以修改类实例化过程
        instance = self.cls(*args, **kwargs)
        return instance

@MyClassDecorator
class MyClass:
    def __init__(self, name):
        self.name = name

# 等价于
MyClass = MyClassDecorator(MyClass)

📌 实战：单例模式

class Singleton:
    """单例装饰器"""
    _instances = {}
    
    def __init__(self, func):
        self.func = func
    
    def __call__(self, *args, **kwargs):
        if self.func not in self._instances:
            self._instances[self.func] = self.func(*args, **kwargs)
        return self._instances[self.func]

@Singleton
class Database:
    def __init__(self):
        print("数据库连接建立")
    
    def query(self, sql):
        print(f"执行: {sql}")

db1 = Database()
db2 = Database()
print(db1 is db2)  # True，同一个实例

📌 实战：缓存装饰器

import functools

def memoize(func):
    """记忆化缓存装饰器"""
    cache = {}
    
    @functools.wraps(func)
    def wrapper(*args):
        if args not in cache:
            cache[args] = func(*args)
        return cache[args]
    
    wrapper.cache = cache  # 暴露缓存
    return wrapper

@memoize
def fibonacci(n):
    if n < 2:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

print(fibonacci(100))  # 很快！
print(f"缓存大小: {len(fibonacci.cache)}")

8️⃣ 装饰器工厂

📌 带参数的类装饰器

class ValidityChecker:
    """参数验证装饰器工厂"""
    def __init__(self, **constraints):
        self.constraints = constraints
    
    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            # 验证参数
            for name, constraint in self.constraints.items():
                if name in kwargs:
                    value = kwargs[name]
                    if 'min' in constraint and value < constraint['min']:
                        raise ValueError(f"{name} 不能小于 {constraint['min']}")
                    if 'max' in constraint and value > constraint['max']:
                        raise ValueError(f"{name} 不能大于 {constraint['max']}")
            return func(*args, **kwargs)
        return wrapper

@ValidityChecker(age=(18, 120), score=(0, 100))
def register(name, age, score):
    print(f"{name} 注册成功: 年龄{age}, 分数{score}")

register("Alice", age=25, score=95)
# register("Bob", age=15, score=80)  # ValueError

🔑 常见装饰器应用场景

📌 1. 日志记录

import functools
import logging

def logged(func=None, *, level=logging.DEBUG):
    """日志装饰器"""
    def decorator(f):
        @functools.wraps(f)
        def wrapper(*args, **kwargs):
            logging.log(level, f"调用 {f.__name__}")
            try:
                result = f(*args, **kwargs)
                logging.log(level, f"{f.__name__} 返回 {result}")
                return result
            except Exception as e:
                logging.error(f"{f.__name__} 抛出异常: {e}")
                raise
        return wrapper
    
    if func is None:
        return decorator
    else:
        return decorator(func)

@logged
def divide(a, b):
    return a / b

@logged(level=logging.WARNING)
def risky_function():
    pass

📌 2. 权限验证

class User:
    def __init__(self, name, role):
        self.name = name
        self.role = role

def require_permission(*permissions):
    """权限检查装饰器"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(user, *args, **kwargs):
            if not hasattr(user, 'role'):
                raise PermissionError("无效用户")
            for perm in permissions:
                if perm not in get_permissions(user.role):
                    raise PermissionError(f"需要权限: {perm}")
            return func(user, *args, **kwargs)
        return wrapper
    return decorator

def get_permissions(role):
    permissions = {
        'admin': ['read', 'write', 'delete', 'admin'],
        'editor': ['read', 'write'],
        'viewer': ['read'],
    }
    return permissions.get(role, [])

@require_permission('write')
def edit_content(user):
    print(f"{user.name} 编辑内容")

admin = User("Admin", "admin")
editor = User("Editor", "editor")

edit_content(admin)   # 成功
edit_content(editor)  # 成功

# edit_content(viewer)  # PermissionError

📌 3. 重试机制

import functools
import time

def retry(max_attempts=3, delay=1, exceptions=(Exception,)):
    """重试装饰器"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    if attempt == max_attempts - 1:
                        raise
                    print(f"尝试 {attempt + 1} 失败: {e}")
                    time.sleep(delay)
            return None
        return wrapper
    return decorator

@retry(max_attempts=3, delay=2, exceptions=(ConnectionError,))
def fetch_data():
    # 模拟可能失败的网络请求
    import random
    if random.random() < 0.7:
        raise ConnectionError("网络连接失败")
    return "数据获取成功"

📌 4. 性能监控

import functools
import time

def performance_monitor(func):
    """性能监控装饰器"""
    call_count = 0
    total_time = 0
    
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        nonlocal call_count, total_time
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        
        call_count += 1
        total_time += elapsed
        
        print(f"{func.__name__}: 耗时 {elapsed*1000:.2f}ms "
              f"(累计 {call_count} 次, 平均 {total_time/call_count*1000:.2f}ms)")
        return result
    
    wrapper.stats = lambda: (call_count, total_time)
    return wrapper

@performance_monitor
def heavy_computation():
    time.sleep(0.1)
    return 42

for _ in range(5):
    heavy_computation()
