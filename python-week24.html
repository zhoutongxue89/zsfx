# todo.py - 命令行TODO应用
import sqlite3
import argparse
from datetime import datetime

class TodoApp:
    def __init__(self, db_path='todo.db'):
        self.db = sqlite3.connect(db_path)
        self.create_table()
    
    def create_table(self):
        self.db.execute('''
            CREATE TABLE IF NOT EXISTS todos (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                task TEXT NOT NULL,
                done BOOLEAN DEFAULT 0,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        ''')
        self.db.commit()
    
    def add(self, task):
        self.db.execute('INSERT INTO todos (task) VALUES (?)', (task,))
        self.db.commit()
        print(f'已添加: {task}')
    
    def list(self, done=None):
        cursor = self.db.execute(
            'SELECT id, task, done FROM todos WHERE done=? OR ?=0 ORDER BY created_at',
            (done or 0, done or 0)
        )
        for row in cursor:
            status = '[✓]' if row[2] else '[ ]'
            print(f'{row[0]} {status} {row[1]}')
    
    def done(self, task_id):
        self.db.execute('UPDATE todos SET done=1 WHERE id=?', (task_id,))
        self.db.commit()
        print(f'任务 {task_id} 已完成')
    
    def delete(self, task_id):
        self.db.execute('DELETE FROM todos WHERE id=?', (task_id,))
        self.db.commit()
        print(f'任务 {task_id} 已删除')

3️⃣ 项目：RESTful API服务

# api.py - Flask RESTful API
from flask import Flask, request, jsonify
from flask_restful import Resource, Api
from flask_sqlalchemy import SQLAlchemy
from flask_jwt import JWT, jwt_required, current_identity
from datetime import datetime

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///app.db'
app.config['SECRET_KEY'] = 'secret'
db = SQLAlchemy(app)
api = Api(app)
jwt = JWT(app)

# 模型
class Article(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(200), nullable=False)
    content = db.Column(db.Text)
    author_id = db.Column(db.Integer)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

# API资源
class ArticleListAPI(Resource):
    def get(self):
        articles = Article.query.order_by(Article.created_at.desc()).all()
        return {'articles': [a.to_dict() for a in articles]}
    
    @jwt_required()
    def post(self):
        data = request.get_json()
        article = Article(title=data['title'], content=data['content'])
        db.session.add(article)
        db.session.commit()
        return article.to_dict(), 201

class ArticleAPI(Resource):
    def get(self, article_id):
        article = Article.query.get_or_404(article_id)
        return article.to_dict()
    
    @jwt_required()
    def put(self, article_id):
        article = Article.query.get_or_404(article_id)
        data = request.get_json()
        article.title = data.get('title', article.title)
        article.content = data.get('content', article.content)
        db.session.commit()
        return article.to_dict()
    
    @jwt_required()
    def delete(self, article_id):
        article = Article.query.get_or_404(article_id)
        db.session.delete(article)
        db.session.commit()
        return '', 204

api.add_resource(ArticleListAPI, '/api/articles')
api.add_resource(ArticleAPI, '/api/articles/')

if __name__ == '__main__':
    with app.app_context():
        db.create_all()
    app.run(debug=True)

4️⃣ 项目：数据可视化看板

# dashboard.py - 数据可视化看板
import pandas as pd
import plotly.express as px
import plotly.graph_objects as go
from dash import Dash, dcc, html
from dash.dependencies import Input, Output

# 准备数据
df = pd.read_csv('sales_data.csv')
df['date'] = pd.to_datetime(df['date'])

# 创建Dash应用
app = Dash(__name__)

app.layout = html.Div([
    html.H1('销售数据看板'),
    
    # 筛选器
    html.Div([
        html.Label('选择日期范围'),
        dcc.DatePickerRange(
            id='date-range',
            start_date=df['date'].min(),
            end_date=df['date'].max()
        ),
        html.Label('选择地区'),
        dcc.Dropdown(
            id='region-selector',
            options=[{'label': r, 'value': r} for r in df['region'].unique()],
            multi=True
        )
    ]),
    
    # 图表
    dcc.Graph(id='sales-trend'),
    dcc.Graph(id='category-pie'),
    dcc.Graph(id='regional-bar')
])

@app.callback(
    [Output('sales-trend', 'figure'),
     Output('category-pie', 'figure'),
     Output('regional-bar', 'figure')],
    [Input('date-range', 'start_date'),
     Input('date-range', 'end_date'),
     Input('region-selector', 'value')]
)
def update_charts(start, end, regions):
    filtered = df[(df['date'] >= start) & (df['date'] <= end)]
    if regions:
        filtered = filtered[filtered['region'].isin(regions)]
    
    # 趋势图
    trend = filtered.groupby('date')['revenue'].sum().reset_index()
    fig1 = px.line(trend, x='date', y='revenue', title='销售趋势')
    
    # 饼图
    fig2 = px.pie(filtered, values='revenue', names='category', title='分类占比')
    
    # 柱状图
    fig3 = px.bar(filtered.groupby('region')['revenue'].sum().reset_index(),
                  x='region', y='revenue', title='地区销售')
    
    return fig1, fig2, fig3

if __name__ == '__main__':
    app.run_server(debug=True)
