1️⃣ 进程 vs 线程

🔑 核心概念
  • 进程（Process）：操作系统分配的独立资源单位，拥有独立的内存空间
  • 线程（Thread）：进程内的执行单元，共享进程资源

🧵 线程

  • 轻量级，创建销毁开销小
  • 共享进程内存空间
  • 受GIL限制（Python）
  • 一个线程崩溃可能影响整个进程
  • 适合I/O密集型任务

⚙️ 进程

  • 重量级，独立内存空间
  • 进程间数据隔离
  • 绕过GIL，真正并行
  • 进程相互独立，更安全
  • 适合CPU密集型任务

2️⃣ multiprocessing模块

multiprocessing是Python标准库中用于多进程编程的核心模块，提供了与threading类似的API。

📌 为什么需要multiprocessing？

import threading
import time

def cpu_task():
    """CPU密集型任务"""
    total = 0
    for i in range(50_000_000):
        total += i ** 0.5
    return total

# 多线程 - 受GIL限制
start = time.time()
threads = [threading.Thread(target=cpu_task) for _ in range(4)]
for t in threads: t.start()
for t in threads: t.join()
print(f"多线程: {time.time() - start:.2f}s")  # 约8秒

# 使用multiprocessing
import multiprocessing

def cpu_task_mp():
    total = 0
    for i in range(50_000_000):
        total += i ** 0.5
    return total

start = time.time()
processes = [multiprocessing.Process(target=cpu_task_mp) for _ in range(4)]
for p in processes: p.start()
for p in processes: p.join()
print(f"多进程: {time.time() - start:.2f}s")  # 约2秒（4核并行）
✅ GIL与多进程

每个Python进程有独立的Python解释器和独立的GIL，真正实现并行计算！

3️⃣ 进程创建与启动

📌 方法一：直接创建

import multiprocessing
import time

def worker(name):
    print(f"进程 {name} 开始")
    time.sleep(2)
    print(f"进程 {name} 完成")

if __name__ == "__main__":
    # 创建进程
    p = multiprocessing.Process(target=worker, args=("Worker-1",))
    
    # 启动进程
    p.start()
    
    # 等待进程结束
    p.join()
    
    print("主进程结束")

📌 方法二：继承Process类

import multiprocessing
import time

class MyProcess(multiprocessing.Process):
    def __init__(self, name, seconds):
        super().__init__()
        self.name = name
        self.seconds = seconds
    
    def run(self):
        print(f"进程 {self.name} 开始")
        time.sleep(self.seconds)
        print(f"进程 {self.name} 完成")
        return self.name

if __name__ == "__main__":
    processes = []
    for i in range(3):
        p = MyProcess(f"Worker-{i}", 2-i)
        processes.append(p)
        p.start()
    
    # 等待所有进程
    for p in processes:
        p.join()
    
    print("全部完成")

📌 方法三：使用上下文

import multiprocessing as mp

# Windows上推荐使用spawn方式（更安全）
if __name__ == "__main__":
    ctx = mp.get_context('spawn')  # 或 'fork' (Unix), 'forkserver'
    
    def worker(q):
        q.put("Hello from child")
    
    queue = ctx.Queue()
    p = ctx.Process(target=worker, args=(queue,))
    p.start()
    print(queue.get())  # Hello from child
    p.join()
⚠️ Windows注意事项

Windows没有fork系统调用，必须使用spawn或forkserver方式，且所有代码必须放在if __name__ == "__main__":下！

4️⃣ 进程间通信

🔑 进程间通信（IPC）

进程拥有独立内存空间，不能直接共享变量，需要通过IPC机制通信。

📌 Queue队列

import multiprocessing as mp

def producer(q):
    for i in range(5):
        q.put(i)
    q.put(None)  # 发送结束信号

def consumer(q):
    while True:
        item = q.get()
        if item is None:
            break
        print(f"消费: {item}")

if __name__ == "__main__":
    queue = mp.Queue()
    
    p1 = mp.Process(target=producer, args=(queue,))
    p2 = mp.Process(target=consumer, args=(queue,))
    
    p1.start()
    p2.start()
    
    p1.join()
    p2.join()
    
    print("完成")

📌 Pipe管道

import multiprocessing as mp

def sender(conn):
    conn.send("Hello from sender")
    conn.send([1, 2, 3])
    conn.close()

def receiver(conn):
    msg1 = conn.recv()
    msg2 = conn.recv()
    print(f"收到: {msg1}")
    print(f"收到: {msg2}")
    conn.close()

if __name__ == "__main__":
    # 创建管道，返回(conn1, conn2)
    parent_conn, child_conn = mp.Pipe()
    
    p1 = mp.Process(target=sender, args=(child_conn,))
    p2 = mp.Process(target=receiver, args=(parent_conn,))
    
    p1.start()
    p2.start()
    
    p1.join()
    p2.join()
💡 Queue vs Pipe
  • Queue：适合多生产者多消费者，支持多进程
  • Pipe：适合一对一通信，速度更快

📌 Manager共享数据

import multiprocessing as mp

def worker(shared_dict, shared_list):
    # 通过Manager共享字典和列表
    shared_dict['count'] = shared_dict.get('count', 0) + 1
    shared_list.append(shared_dict['count'])

if __name__ == "__main__":
    with mp.Manager() as manager:
        shared_dict = manager.dict()
        shared_list = manager.list()
        
        processes = []
        for i in range(5):
            p = mp.Process(target=worker, args=(shared_dict, shared_list))
            processes.append(p)
            p.start()
        
        for p in processes:
            p.join()
        
        print(f"共享字典: {dict(shared_dict)}")
        print(f"共享列表: {list(shared_list)}")

5️⃣ 进程池Pool

🔑 为什么需要进程池？

进程创建销毁开销大，进程池复用进程，高效管理并发数量。

📌 Pool基本用法

import multiprocessing as mp
import time

def cpu_task(n):
    """模拟CPU密集型任务"""
    return sum(i ** 2 for i in range(n))

if __name__ == "__main__":
    # 创建进程池（默认使用所有CPU核心）
    with mp.Pool(processes=4) as pool:
        # 方法1: map - 按顺序返回结果
        results = pool.map(cpu_task, [1000000, 2000000, 3000000, 4000000])
        print(f"map结果: {results}")
        
        # 方法2: apply_async - 异步提交单个任务
        result = pool.apply_async(cpu_task, (5000000,))
        print(f"异步结果: {result.get()}")  # 等待结果
        
        # 方法3: apply_async + 回调
        def callback(result):
            print(f"任务完成，结果: {result}")
        
        pool.apply_async(cpu_task, (1000000,), callback=callback)
        
        # 方法4: starmap - 多参数
        data = [(1, 2), (3, 4), (5, 6)]
        results = pool.starmap(lambda x, y: x + y, data)
        print(f"starmap结果: {results}")

📌 imap与imap_unordered

import multiprocessing as mp
import time

def task(n):
    time.sleep(0.5)
    return n * n

if __name__ == "__main__":
    with mp.Pool(4) as pool:
        # imap - 按顺序返回（generator）
        print("imap（按顺序）:")
        for result in pool.imap(task, range(8)):
            print(f"  结果: {result}")
        
        # imap_unordered - 谁先完成谁先返回
        print("\nimap_unordered（按完成顺序）:")
        for result in pool.imap_unordered(task, range(8)):
            print(f"  结果: {result}")

6️⃣ 共享内存与锁

📌 Value与Array

import multiprocessing as mp

def worker(shared_counter, shared_array, lock):
    """使用共享内存"""
    with lock:
        shared_counter.value += 1
        shared_array[shared_counter.value % len(shared_array)] = shared_counter.value

if __name__ == "__main__":
    # 创建共享内存
    counter = mp.Value('i', 0)  # 有符号整数
    array = mp.Array('i', range(5))  # 整数数组
    lock = mp.Lock()  # 进程锁
    
    print(f"初始: counter={counter.value}, array={list(array)}")
    
    processes = []
    for i in range(4):
        p = mp.Process(target=worker, args=(counter, array, lock))
        processes.append(p)
        p.start()
    
    for p in processes:
        p.join()
    
    print(f"最终: counter={counter.value}, array={list(array)}")
💡 类型代码
  • 'i'：有符号整数
  • 'I'：无符号整数
  • 'd'：双精度浮点数
  • 'c'：字符（bytes）

📌 进程间同步

import multiprocessing as mp
import time

def producer(queue, event):
    for i in range(5):
        queue.put(i)
        print(f"生产: {i}")
    event.set()  # 发送完成信号

def consumer(queue, event):
    event.wait()  # 等待完成信号
    while not queue.empty():
        item = queue.get()
        print(f"消费: {item}")

if __name__ == "__main__":
    queue = mp.Queue()
    event = mp.Event()
    
    p1 = mp.Process(target=producer, args=(queue, event))
    p2 = mp.Process(target=consumer, args=(queue, event))
    
    p1.start()
    p2.start()
    
    p1.join()
    p2.join()

7️⃣ 进程 vs 线程选择

场景 推荐 原因
Web服务器 多进程（nginx/uwsgi） 隔离性好，一个进程崩溃不影响其他
网络爬虫 多线程（I/O等待长） GIL在等待时可释放，开销小
图像处理 多进程 CPU密集，绕过GIL
科学计算 多进程 / numpy向量化 numpy底层已优化
数据库操作 多线程 + 连接池 主要是I/O等待
机器学习 多进程 / GPU 训练是CPU/GPU密集型

📌 混合使用（进程+线程）

import multiprocessing as mp
import threading
import queue

def parallel_processing(data):
    """
    多进程处理数据 + 每进程内多线程处理I/O
    适合：CPU计算 + I/O操作混合场景
    """
    def process_chunk(chunk_id, data_chunk, result_queue):
        """处理单个数据块（进程函数）"""
        local_results = []
        
        def fetch_external(item):
            # 每个进程内可以用多线程做I/O
            import time
            time.sleep(0.1)  # 模拟网络请求
            return f"processed-{item}"
        
        # 使用线程池处理I/O
        from concurrent.futures import ThreadPoolExecutor
        with ThreadPoolExecutor(max_workers=4) as executor:
            futures = [executor.submit(fetch_external, item) for item in data_chunk]
            for future in futures:
                local_results.append(future.result())
        
        result_queue.put((chunk_id, local_results))
    
    # 分片数据
    chunk_size = len(data) // 4
    chunks = [data[i:i+chunk_size] for i in range(0, len(data), chunk_size)]
    
    result_queue = mp.Queue()
    processes = []
    
    for i, chunk in enumerate(chunks):
        p = mp.Process(target=process_chunk, args=(i, chunk, result_queue))
        processes.append(p)
        p.start()
    
    # 收集结果
    results = []
    for _ in range(len(chunks)):
        chunk_id, chunk_results = result_queue.get()
        results.append((chunk_id, chunk_results))
    
    for p in processes:
        p.join()
    
    # 按顺序整理结果
    results.sort(key=lambda x: x[0])
    return results

if __name__ == "__main__":
    data = list(range(20))
    results = parallel_processing(data)
    print(f"处理完成，共 {len(results)} 个数据块")

8️⃣ 实战：并行计算

🔑 项目目标

使用多进程并行计算素数，演示如何加速CPU密集型任务。

import multiprocessing as mp
import time
import math

def is_prime(n):
    """判断是否为素数"""
    if n < 2:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    for i in range(3, int(math.sqrt(n)) + 1, 2):
        if n % i == 0:
            return False
    return True

def find_primes_in_range(start, end):
    """在指定范围内找素数"""
    primes = []
    for n in range(start, end):
        if is_prime(n):
            primes.append(n)
    return primes

def parallel_prime_finder(limit, num_processes=None):
    """并行找素数"""
    if num_processes is None:
        num_processes = mp.cpu_count()
    
    # 生成分段
    chunk_size = limit // num_processes
    ranges = [(i * chunk_size, (i + 1) * chunk_size) for i in range(num_processes)]
    ranges[-1] = (ranges[-1][0], limit + 1)  # 确保包含最后一位
    
    # 并行处理
    with mp.Pool(processes=num_processes) as pool:
        results = pool.starmap(find_primes_in_range, ranges)
    
    # 合并结果
    all_primes = []
    for primes in results:
        all_primes.extend(primes)
    
    return sorted(all_primes)

if __name__ == "__main__":
    LIMIT = 2_000_000
    
    # 单进程
    start = time.time()
    primes_single = find_primes_in_range(2, LIMIT)
    single_time = time.time() - start
    print(f"单进程: 找到 {len(primes_single)} 个素数，耗时 {single_time:.2f}s")
    
    # 多进程
    start = time.time()
    primes_parallel = parallel_prime_finder(LIMIT)
    parallel_time = time.time() - start
    print(f"多进程: 找到 {len(primes_parallel)} 个素数，耗时 {parallel_time:.2f}s")
    
    # 加速比
    print(f"加速比: {single_time / parallel_time:.2f}x")

9️⃣ 综合项目：多进程爬虫

import multiprocessing as mp
import time
import os
from urllib.request import urlopen
from urllib.error import URLError
import hashlib

class MultiProcessCrawler:
    def __init__(self, max_workers=4):
        self.max_workers = max_workers
    
    def fetch_url(self, url):
        """获取单个URL内容"""
        try:
            with urlopen(url, timeout=10) as response:
                content = response.read()
                return {
                    "status": "success",
                    "url": url,
                    "content": content,
                    "size": len(content)
                }
        except Exception as e:
            return {
                "status": "error",
                "url": url,
                "error": str(e)
            }
    
    def process_url(self, args):
        """处理单个URL（进程函数）"""
        url, output_dir = args
        
        result = self.fetch_url(url)
        
        if result["status"] == "success":
            # 生成文件名
            filename = hashlib.md5(url.encode()).hexdigest() + ".html"
            filepath = os.path.join(output_dir, filename)
            
            with open(filepath, 'wb') as f:
                f.write(result["content"])
            
            result["saved_as"] = filepath
        
        return result
    
    def crawl(self, urls, output_dir="output"):
        """并发爬取所有URL"""
        os.makedirs(output_dir, exist_ok=True)
        
        tasks = [(url, output_dir) for url in urls]
        
        results = []
        with mp.Pool(processes=self.max_workers) as pool:
            for result in pool.imap_unordered(self.process_url, tasks):
                results.append(result)
                if result["status"] == "success":
                    print(f"✓ {result['url']} -> {result.get('saved_as', 'N/A')}")
                else:
                    print(f"✗ {result['url']} - {result['error']}")
        
        return results

if __name__ == "__main__":
    # 示例URL
    urls = [
        "https://example.com",
        "https://httpbin.org/html",
        "https://httpbin.org/delay/1",
    ]
    
    crawler = MultiProcessCrawler(max_workers=3)
    results = crawler.crawl(urls, output_dir="crawled_pages")
    
    # 统计
    success = sum(1 for r in results if r["status"] == "success")
    print(f"\n爬取完成: 成功 {success}/{len(results)}")
