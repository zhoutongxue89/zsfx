字符串基础回顾

字符串（str）是 Python 中最常用的数据类型，存储文本数据。字符串是不可变的有序序列。

# 创建字符串的各种方式
s1 = '单引号'
s2 = "双引号"
s3 = """三引号
可以换行"""
s4 = r"C:\Users\test"   # 原始字符串，\不转义
s5 = f"2+2={2+2}"         # f-string（Python 3.6+）
s6 = b"bytes string"       # 字节字符串

# 字符串拼接
first = "Hello"
last = "World"
full = first + ", " + last + "!"  # Hello, World!

# 字符串重复
print("Ha" * 3)   # HaHaHa
print("=" * 40)  # 40个等号

# 索引和切片（和列表完全一样）
text = "Python"
print(text[0])     # P
print(text[-1])    # n
print(text[:3])    # Pyt
print(text[::-1])  # nohtyP（反转）

# 字符串是不可变的！
# text[0] = "J"  # 报错！TypeError

字符串方法大全（25+个）

大小写转换

upper()

全部大写

"hello".upper() → "HELLO"
lower()

全部小写

"HELLO".lower() → "hello"
title()

每个单词首字母大写

"hello world".title() → "Hello World"
capitalize()

首字母大写

"hello world".capitalize() → "Hello world"
swapcase()

大小写互换

"Hello".swapcase() → "hELLO"

查找与替换

find(sub)

找子串，返回索引，不存在返回-1

"hello".find("ll") → 2
index(sub)

找子串，不存在抛出ValueError

"hello".index("e") → 1
count(sub)

统计子串出现次数

"banana".count("a") → 3
replace(old, new)

替换子串

"a-b-c".replace("-", "/") → "a/b/c"
startswith(prefix)

是否以prefix开头

"Python".startswith("Py") → True
endswith(suffix)

是否以suffix结尾

"test.py".endswith(".py") → True

空白处理

strip()

去除两端空白字符

" hello ".strip() → "hello"
lstrip()

去除左端空白

" hello ".lstrip() → "hello "
rstrip()

去除右端空白

" hello ".rstrip() → " hello"

分割与合并

split(sep)

分割字符串为列表

"a,b,c".split(",") → ["a","b","c"]
splitlines()

按换行符分割

"a\nb\nc".splitlines() → ["a","b","c"]
join(iterable)

用字符串连接序列

",".join(["a","b","c"]) → "a,b,c"

是否全为数字

" ".isspace() → True

3

格式化输出（3种方式）

name = "周同学"
score = 98.5
rank = 1

# 方式1：% 格式化（老写法，了解即可）
print("姓名：%s，成绩：%.1f，排名：%d" % (name, score, rank))

# 方式2：str.format()（Python 2.7+）
print("姓名：{}，成绩：{:.1f}，排名：{}".format(name, score, rank))
print("姓名：{name}，成绩：{score:.2f}".format(name=name, score=score))

# 方式3：f-string（Python 3.6+，推荐！）
print(f"姓名：{name}，成绩：{score:.1f}，排名：{rank}")
print(f"成绩是否优秀：{score >= 90}")  # 可以放表达式
print(f"{name!r}")                    # 带引号的repr

# f-string 格式说明符
pi = 3.14159265
print(f"{pi:.2f}")      # 3.14（两位小数）
print(f"{pi:.4f}")      # 3.1416（四位小数）
print(f"{pi:10.2f}")    # '      3.14'（宽度10，右对齐）
print(f"{pi:<10.2f}")   # '3.14      '（左对齐）

num = 1234567
print(f"{num:,}")        # 1,234,567（千位分隔符）
print(f"{0.85:.1%}")     # 85.0%（百分比）
print(f"{255:08b}")      # 11111111（二进制，8位补零）
print(f"{255:02x}")      # ff（十六进制）

正则表达式入门

正则表达式是处理文本的超级工具，用特殊语法匹配字符串模式。

import re

text = "我的电话是 138-1234-5678，邮箱是 test@example.com"

# re.search()：搜索第一个匹配
phone_pattern = r"\d{3}-\d{4}-\d{4}"
match = re.search(phone_pattern, text)
if match:
    print(f"电话号码：{match.group()}")  # 138-1234-5678

# re.findall()：找所有匹配
numbers = re.findall(r"\d+", text)
print(f"所有数字：{numbers}")  # ['138', '1234', '5678']

# re.sub()：替换
censored = re.sub(r"\d{3}-\d{4}-\d{4}", "***-****-****", text)
print(censored)

# 常用正则语法
# \d  → 数字 [0-9]
# \w  → 单词字符 [a-zA-Z0-9_]
# \s  → 空白字符
# .   → 任意字符（除换行）
# +   → 1个或多个
# *   → 0个或多个
# ?   → 0个或1个
# {n} → 恰好n个
# ^   → 开头
# $   → 结尾

# 验证手机号
def is_valid_phone(phone):
    pattern = r"^1[3-9]\d{9}$"
    return bool(re.match(pattern, phone))

print(is_valid_phone("13812345678"))  # True
print(is_valid_phone("12345"))         # False

文件读写基础

文件打开模式

"r"

只读
文件必须存在

"w"

只写
覆盖（创建新文件）

"a"

追加
在末尾添加

"r+"

读写
文件必须存在

# 写入文件
file = open("test.txt", "w", encoding="utf-8")
file.write("第一行内容\n")
file.write("第二行内容\n")
file.close()  # 必须关闭！

# 读取整个文件
file = open("test.txt", "r", encoding="utf-8")
content = file.read()
file.close()
print(content)

# 逐行读取
file = open("test.txt", "r", encoding="utf-8")
for line in file:
    print(line.rstrip())  # rstrip去掉行尾换行符
file.close()

# 读取所有行为列表
file = open("test.txt", "r", encoding="utf-8")
lines = file.readlines()
file.close()
print(lines)  # ['第一行内容\n', '第二行内容\n']
⚠️ 手动 open 必须手动 close！否则文件句柄泄漏。推荐用 with 语句（下节）。

with 语句（推荐写法）

with 语句会自动关闭文件，即使发生异常也不会忘记关闭！

# 写文件
with open("notes.txt", "w", encoding="utf-8") as f:
    f.write("Python 学习笔记\n")
    f.write("第八周：文件操作\n")
    f.writelines(["字符串处理\n", "正则表达式\n"])
# 出了 with 块，文件自动关闭

# 读文件
with open("notes.txt", "r", encoding="utf-8") as f:
    content = f.read()
    print(content)

# 追加写入
with open("notes.txt", "a", encoding="utf-8") as f:
    f.write("文件操作完成！\n")

# 同时打开多个文件（Python 3.1+）
with open("input.txt", "r", encoding="utf-8") as fin, \
     open("output.txt", "w", encoding="utf-8") as fout:
    for line in fin:
        fout.write(line.upper())  # 全部转大写写入output
💡 encoding 参数：Windows 默认编码是 GBK，如果文件包含中文，必须指定 encoding="utf-8"，否则会乱码！

文件路径处理（pathlib）

from pathlib import Path
import os

# 推荐：使用 pathlib（Python 3.4+）
p = Path("my_folder/data.txt")

# 路径信息
print(p.name)        # data.txt（文件名）
print(p.stem)        # data（不含扩展名）
print(p.suffix)      # .txt（扩展名）
print(p.parent)      # my_folder（父目录）
print(p.is_file())   # 是否是文件
print(p.is_dir())    # 是否是目录
print(p.exists())    # 是否存在

# 路径拼接（用 / 运算符！跨平台安全）
base = Path("C:/Users/zhou")
file_path = base / "documents" / "notes.txt"
print(file_path)  # C:\Users\zhou\documents\notes.txt

# 创建目录
Path("new_folder").mkdir(exist_ok=True)

# 遍历目录下的所有 .txt 文件
for txt_file in Path(".").glob("*.txt"):
    print(txt_file)

# 读写文件（pathlib 自带）
p = Path("test.txt")
p.write_text("hello world", encoding="utf-8")

print(content)

CSV 文件处理

CSV（逗号分隔值）是最常见的数据文件格式，Excel 可以直接打开。

import csv

# 写入 CSV
students = [
    ["姓名", "成绩", "城市"],
    ["小明", 85, "北京"],
    ["小红", 92, "上海"],
    ["小李", 78, "广州"],
]

with open("students.csv", "w", newline="", encoding="utf-8-sig") as f:
    writer = csv.writer(f)
    writer.writerows(students)

# 读取 CSV
with open("students.csv", "r", encoding="utf-8-sig") as f:
    reader = csv.reader(f)
    header = next(reader)  # 读取标题行
    print(f"列名：{header}")
    for row in reader:
        print(row)

# 使用 DictReader（更方便）
with open("students.csv", "r", encoding="utf-8-sig") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(f"{row['姓名']}的成绩：{row['成绩']}")
💡 Excel中文：用 encoding="utf-8-sig" 可以避免 Excel 打开 CSV 时中文乱码！

🎯 实战1：日记本程序

from pathlib import Path
from datetime import datetime

DIARY_FILE = Path("my_diary.txt")

def write_diary(content):
    """写入一篇日记（追加模式）"""
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    entry = f"\n{'='*50}\n📅 {timestamp}\n{'='*50}\n{content}\n"
    with open(DIARY_FILE, "a", encoding="utf-8") as f:
        f.write(entry)
    print("✅ 日记已保存！")

def read_diary():
    """读取所有日记"""
    if not DIARY_FILE.exists():
        print("📭 还没有写过日记哦！")
        return
    with open(DIARY_FILE, "r", encoding="utf-8") as f:
        print(f.read())

def search_diary(keyword):
    """搜索包含关键词的日记"""
    if not DIARY_FILE.exists():
        print("没有日记文件")
        return
    with open(DIARY_FILE, "r", encoding="utf-8") as f:
        content = f.read()
    count = content.count(keyword)
    print(f"🔍 找到 '{keyword}' 共 {count} 次")

# 测试
write_diary("今天学了Python文件操作，感觉很厉害！明天继续！")
write_diary("完成了第八周的学习，Python基础语法全部搞定啦！")
read_diary()
search_diary("Python")

🎯 实战2：单词频率分析器

import re
from collections import Counter

def analyze_text(text_or_file):
    """分析文本词频，支持字符串或文件路径"""
    from pathlib import Path
    p = Path(text_or_file)
    if p.exists():
        text = p.read_text(encoding="utf-8")
    else:
        text = text_or_file
    
    # 提取英文单词
    words = re.findall(r"[a-zA-Z]+", text.lower())
    
    # 停用词（不统计）
    stopwords = {"the", "a", "an", "is", "are", "was", "in", "to", "of", "and"}
    words = [w for w in words if w not in stopwords and len(w) > 2]
    
    counter = Counter(words)
    total = len(words)
    
    print(f"\n📊 词频分析报告")
    print(f"总单词数：{total}，独特词数：{len(counter)}")
    print(f"\n🏆 TOP 10 高频词：")
    for word, count in counter.most_common(10):
        bar = "▓" * (count * 20 // counter.most_common(1)[0][1])
        print(f"  {word:15s} {count:4d}次 {bar}")

sample = """Python is great for data science and machine learning.
Python makes programming fun and productive.
With Python you can build anything you can imagine."""
analyze_text(sample)
