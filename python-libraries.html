1. pip包管理器

pip简介

pip是Python的包安装器，用于安装和管理Python包。Python 3.4+自带pip。

基础命令

# 查看pip版本
pip --version
pip3 --version

# 升级pip
python -m pip install --upgrade pip

# 安装包
pip install requests
pip install numpy

# 安装指定版本
pip install requests==2.28.0
pip install "requests>=2.28.0"

# 升级包
pip install --upgrade requests

# 卸载包
pip uninstall requests

# 查看已安装的包
pip list

# 导出依赖列表
pip freeze > requirements.txt

# 从文件安装
pip install -r requirements.txt

国内镜像源

国内访问PyPI较慢，建议使用镜像源：

# 临时使用
pip install requests -i https://pypi.tuna.tsinghua.edu.cn/simple

# 永久设置（清华镜像）
pip install pip -U
pip config set global.index-url https://pypi.tuna.tsinghua.edu.cn/simple

# 其他镜像源
# 阿里云: https://mirrors.aliyun.com/pypi/simple/
# 腾讯云: https://mirrors.cloud.tencent.com/pypi/simple/
# 华为云: https://repo.huaweicloud.com/repository/pypi/simple/
💡 推荐：创建一个 pip.conf 文件永久配置镜像： 
# Windows: C:\Users\用户名\pip\pip.ini
# Linux/Mac: ~/.pip/pip.conf

[global]
index-url = https://pypi.tuna.tsinghua.edu.cn/simple
trusted-host = pypi.tuna.tsinghua.edu.cn

2. 虚拟环境

为什么需要虚拟环境？

不同项目可能依赖不同版本的包，虚拟环境可以为每个项目创建独立的Python环境，避免版本冲突。

创建虚拟环境

# 使用venv模块（Python 3内置）
python -m venv myenv

# 指定Python版本（需要已安装）
python3.11 -m venv myenv

# 在当前目录创建
python -m venv .venv

激活虚拟环境

# Windows
myenv\Scripts\activate
# 或
.\myenv\Scripts\Activate.ps1  # PowerShell

# Linux/Mac
source myenv/bin/activate

使用虚拟环境

# 激活后，安装的包只在该环境中有效
pip install requests

# 退出虚拟环境
deactivate

使用virtualenv

# 安装virtualenv
pip install virtualenv

# 创建虚拟环境
virtualenv myenv

# 指定Python版本
virtualenv -p python3.11 myenv

# 激活和使用同上

推荐：使用venv

# 项目目录结构建议
myproject/
├── .venv/              # 虚拟环境
├── src/                # 源代码
├── tests/              # 测试代码
├── requirements.txt    # 依赖列表
└── README.md

3. Web开发

Flask - 轻量级Web框架

特点：简单、灵活、易扩展

# 安装
pip install flask

# 示例代码
from flask import Flask
app = Flask(__name__)

@app.route('/')
def hello():
    return 'Hello, World!'

if __name__ == '__main__':
    app.run(debug=True)

Django - 全功能Web框架

特点：功能完善、ORM内置、管理后台

# 安装
pip install django

# 创建项目
django-admin startproject myproject
cd myproject
python manage.py runserver

FastAPI - 现代Web框架

特点：高性能、自动文档、类型提示

# 安装
pip install fastapi uvicorn

# 示例代码
from fastapi import FastAPI
app = FastAPI()

@app.get('/')
def read_root():
    return {'message': 'Hello, World!'}

# 运行：uvicorn main:app --reload

requests - HTTP请求库

# 安装
pip install requests

# 示例
import requests

# GET请求
response = requests.get('https://api.github.com')
print(response.json())

# POST请求
data = {'key': 'value'}
response = requests.post('https://httpbin.org/post', json=data)

# 带参数
params = {'page': 1, 'per_page': 10}
response = requests.get('https://api.example.com/items', params=params)

4. 数据分析

NumPy - 数值计算

特点：高性能多维数组库，线性代数计算

# 安装
pip install numpy

# 示例
import numpy as np

# 创建数组
arr = np.array([1, 2, 3, 4, 5])
print(arr * 2)  # [2, 4, 6, 8, 10]

# 矩阵运算
matrix = np.array([[1, 2], [3, 4]])
print(np.dot(matrix, matrix))

# 基本统计

filtered = df[df['age'] > 25]

PyTorch - 深度学习

# 安装
pip install torch

# 示例
import torch

# 创建张量
x = torch.tensor([1.0, 2.0, 3.0])
y = torch.tensor([2.0, 4.0, 6.0])

# 定义模型
model = torch.nn.Linear(1, 1)
optimizer = torch.optim.SGD(model.parameters(), lr=0.01)

# 训练
for epoch in range(100):
    pred = model(x.reshape(-1, 1))
    loss = torch.nn.functional.mse_loss(pred, y.reshape(-1, 1))
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()

JSON - JSON处理

import json

# Python对象转JSON字符串
data = {'name': '张三', 'age': 18}
json_str = json.dumps(data, ensure_ascii=False, indent=2)

# JSON字符串转Python对象
data = json.loads(json_str)

# 读写JSON文件
with open('data.json', 'w', encoding='utf-8') as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

with open('data.json', 'r', encoding='utf-8') as f:
    data = json.load(f)

Datetime - 日期时间

from datetime import datetime, timedelta

# 获取当前时间
now = datetime.now()

# 格式化
print(now.strftime('%Y-%m-%d %H:%M:%S'))
print(now.strftime('%Y年%m月%d日'))

# 解析字符串
date = datetime.strptime('2024-01-01', '%Y-%m-%d')

# 时间计算
tomorrow = now + timedelta(days=1)
yesterday = now - timedelta(days=1)

Pathlib - 路径处理

from pathlib import Path

# 路径操作
p = Path('C:/Users/test/documents/file.txt')

print(p.name)      # file.txt
print(p.stem)      # file
print(p.suffix)     # .txt
print(p.parent)     # C:/Users/test/documents

# 创建目录
Path('new_folder').mkdir(exist_ok=True)

# 遍历文件
for f in Path('.').glob('*.py'):
    print(f)

Urllib - URL处理

from urllib.parse import urlparse, urlencode, quote

# URL解析
url = 'https://example.com/path?param=value'
parsed = urlparse(url)
print(parsed.scheme, parsed.netloc, parsed.path)

# URL编码
query = urlencode({'name': '张三', 'age': 18})
print(quote('Hello World'))

# 下载文件
from urllib.request import urlretrieve
urlretrieve('https://example.com/file.txt', 'local_file.txt')
