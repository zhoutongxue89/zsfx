3️⃣ Selenium动态页面

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC

driver = webdriver.Chrome()
driver.get('https://example.com')

# 查找元素
el = driver.find_element(By.ID, 'username')
el = driver.find_element(By.CSS_SELECTOR, 'div.content p')
el.send_keys('搜索内容')  # 输入
el.click()               # 点击

# 等待加载
wait = WebDriverWait(driver, 10)
el = wait.until(EC.presence_of_element_located((By.ID, 'result')))

# 获取数据
html = driver.page_source
text = el.text
driver.quit()

6️⃣ 实战：完整爬虫

class BookSpider:
    def __init__(self):
        self.session = requests.Session()
        self.session.headers = {'User-Agent':'Mozilla/5.0 ...'}
        self.data = []
    
    def scrape_page(self, page):
        url = f'{self.base}/catalogue/page-{page}.html'
        r = self.session.get(url, timeout=10)
        soup = BeautifulSoup(r.text, 'lxml')
        for book in soup.select('article.product_pod'):
            self.data.append({
                'title': book.select_one('h3 a')['title'],
                'price': book.select_one('p.price_color').text,
            })
    
    def run(self, max_pages=10):
        for p in range(1, max_pages+1):
            self.scrape_page(p)
            time.sleep(random.uniform(0.5, 1.5))
        pd.DataFrame(self.data).to_csv('books.csv', index=False)
