一对一通信

效率高

1.2 TCP三次握手与四次挥手

💡 三次握手（建立连接）

1. 客户端发送SYN包到服务器
2. 服务器返回SYN+ACK包
3. 客户端发送ACK包，连接建立

💡 四次挥手（断开连接）

1. 主动方发送FIN包
2. 被动方返回ACK包
3. 被动方发送FIN包
4. 主动方返回ACK包，连接断开

2.1 TCP服务器与客户端

TCP服务器 Python 
import socket

def tcp_server(host='127.0.0.1', port=8888):
    """TCP服务器"""
    # 创建socket对象
    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    
    # 绑定地址
    server.bind((host, port))
    
    # 开始监听（最大5个连接）
    server.listen(5)
    print(f"服务器启动，监听 {host}:{port}")
    
    while True:
        # 接受客户端连接
        client_socket, addr = server.accept()
        print(f"收到连接: {addr}")
        
        # 处理请求
        try:
            while True:
                # 接收数据
                data = client_socket.recv(1024)
                if not data:  # 客户端断开
                    break
                
                message = data.decode('utf-8')
                print(f"收到消息: {message}")
                
                # 发送响应
                response = f"服务器收到: {message}"
                client_socket.send(response.encode('utf-8'))
        
        except Exception as e:
            print(f"处理出错: {e}")
        finally:
            client_socket.close()
            print(f"连接关闭: {addr}")

# 启动服务器
tcp_server()
TCP客户端 Python 
import socket

def tcp_client(host='127.0.0.1', port=8888):
    """TCP客户端"""
    client = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    
    # 连接服务器
    client.connect((host, port))
    print(f"已连接服务器 {host}:{port}")
    
    try:
        while True:
            # 发送消息
            msg = input("请输入消息 (quit退出): ")
            if msg.lower() == 'quit':
                break
            
            client.send(msg.encode('utf-8'))
            
            # 接收响应
            response = client.recv(1024)
            print(f"服务器响应: {response.decode('utf-8')}")
    
    except KeyboardInterrupt:
        print("\n客户端退出")
    finally:
        client.close()

2.2 UDP服务器与客户端

UDP通信 Python 
import socket

def udp_server(host='127.0.0.1', port=8889):
    """UDP服务器"""
    server = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)  # SOCK_DGRAM = UDP
    server.bind((host, port))
    print(f"UDP服务器启动，监听 {host}:{port}")
    
    while True:
        data, addr = server.recvfrom(1024)  # 接收数据和地址
        message = data.decode('utf-8')
        print(f"收到 {addr}: {message}")
        
        # 发送响应
        response = f"收到: {message}"
        server.sendto(response.encode('utf-8'), addr)

def udp_client(host='127.0.0.1', port=8889):
    """UDP客户端"""
    client = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
    
    msg = "Hello, UDP Server!"
    client.sendto(msg.encode('utf-8'), (host, port))
    
    data, server_addr = client.recvfrom(1024)
    print(f"服务器响应: {data.decode('utf-8')}")
    
    client.close()
⚠️ TCP vs UDP选择

TCP适合：文件传输、网页访问、邮件收发等需要可靠传输的场景。
UDP适合：实时音视频、DNS查询、在线游戏等允许少量丢包但要求低延迟的场景。

urllib 与 requests

4.1 urllib 发送请求

urllib请求示例 Python 
import urllib.request
import urllib.parse
import json

# GET请求
url = 'https://httpbin.org/get'
with urllib.request.urlopen(url, timeout=10) as response:
    data = json.loads(response.read())
    print(f"URL: {data['url']}")
    print(f"方法: {data['headers']['User-Agent']}")

# GET带参数
params = urllib.parse.urlencode({'name': '张三', 'age': 25})
url = f'https://httpbin.org/get?{params}'
with urllib.request.urlopen(url) as response:
    data = json.loads(response.read())
    print(f"参数: {data['args']}")

# POST请求
data = urllib.parse.urlencode({'username': 'admin', 'password': '123'}).encode()
req = urllib.request.Request(
    'https://httpbin.org/post',
    data=data,
    method='POST'
)
with urllib.request.urlopen(req) as response:
    result = json.loads(response.read())
    print(f"POST结果: {result['form']}")

# 自定义请求头
headers = {
    'User-Agent': 'Mozilla/5.0',
    'Accept': 'application/json',
    'X-Custom-Header': 'PythonClient'
}
req = urllib.request.Request('https://httpbin.org/headers', headers=headers)
with urllib.request.urlopen(req) as response:
    data = json.loads(response.read())
    print(f"请求头: {data['headers']}")

4.3 进阶用法

requests进阶 Python 
import requests

# 会话保持（Session）
session = requests.Session()
session.headers.update({'User-Agent': 'MyApp/1.0'})

# 先登录
session.post('https://httpbin.org/post', data={'user': 'admin'})

# 后续请求自动携带cookie
response = session.get('https://httpbin.org/get')

# 超时设置
response = requests.get('https://httpbin.org/delay/5', timeout=3)  # 3秒超时

# 异常处理
try:
    response = requests.get('https://invalid-url', timeout=5)
    response.raise_for_status()  # 状态码不是2xx会抛出异常
except requests.exceptions.Timeout:
    print("请求超时")
except requests.exceptions.ConnectionError:
    print("连接错误")
except requests.exceptions.HTTPError as e:
    print(f"HTTP错误: {e}")

# 下载文件
response = requests.get('https://httpbin.org/image', stream=True)
with open('image.png', 'wb') as f:
    for chunk in response.iter_content(chunk_size=8192):
        f.write(chunk)
print("文件下载完成")

from bs4 import BeautifulSoup

5.3 实战：简易HTTP客户端

命令行HTTP客户端 Python 
import requests
import sys
import time

def http_client(url, method='GET', headers=None, data=None):
    """简易HTTP客户端"""
    print(f"\n{'='*60}")
    print(f"URL: {url}")
    print(f"Method: {method}")
    print(f"{'='*60}\n")
    
    start_time = time.time()
    
    try:
        # 发送请求
        if method.upper() == 'GET':
            response = requests.get(url, headers=headers, timeout=10)
        elif method.upper() == 'POST':
            response = requests.post(url, headers=headers, data=data, timeout=10)
        else:
            print(f"不支持的方法: {method}")
            return
        
        elapsed = time.time() - start_time
        
        # 输出响应信息
        print(f"状态码: {response.status_code}")
        print(f"响应时间: {elapsed:.3f}s")
        print(f"Content-Type: {response.headers.get('Content-Type', 'N/A')}")
        print(f"Content-Length: {len(response.content)} bytes\n")
        
        print("响应头:")
        for key, value in response.headers.items():
            print(f"  {key}: {value}")
        
        print(f"\n响应体 (前500字符):")
        print("-"*40)
        print(response.text[:500])
        
    except requests.exceptions.Timeout:
        print("请求超时！")
    except requests.exceptions.ConnectionError as e:
        print(f"连接错误: {e}")
    except Exception as e:
        print(f"错误: {e}")

# 使用示例
if __name__ == '__main__':
    if len(sys.argv) < 2:
        print("用法: python http_client.py ")
        print("示例: python http_client.py https://httpbin.org/get")
        sys.exit(1)
    
    url = sys.argv[1]
    http_client(url)
