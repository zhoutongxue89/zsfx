为什么要用函数？

想象你要计算三个圆的面积，如果不用函数：

# ❌ 没有函数（重复代码！）
import math
area1 = math.pi * 3 ** 2
area2 = math.pi * 5 ** 2
area3 = math.pi * 7 ** 2

# ✅ 有了函数（一次定义，多次使用）
def circle_area(radius):
    return math.pi * radius ** 2

area1 = circle_area(3)
area2 = circle_area(5)
area3 = circle_area(7)

函数的定义与调用

语法：
def 函数名(参数1, 参数2, ...):
    """文档字符串（可选）"""
    函数体
    return 返回值（可选） 
# 最简单的函数
def say_hello():
    print("Hello, World!")

say_hello()  # 调用：Hello, World!

# 带参数的函数
def greet(name):
    print(f"你好，{name}！")

greet("周同学")   # 你好，周同学！
greet("Python")  # 你好，Python！

# 带返回值的函数
def add(a, b):
    result = a + b
    return result

print(add(3, 4))   # 7
total = add(10, 20)
print(total)      # 30

# 函数是对象，可以赋值给变量
my_add = add
print(my_add(5, 5))  # 10
💡 命名规范：函数名用小写字母+下划线（snake_case），动词开头，如 get_name()、calculate_tax()、is_valid()

参数详解（4种类型）

① 位置参数

按顺序传递，最基础的形式

② 默认参数

有默认值，不传则用默认

③ *args 可变位置

接收任意数量的位置参数

④ **kwargs 可变关键字

接收任意数量的关键字参数

① 位置参数 & 关键字传参

def describe_person(name, age, city):
    print(f"{name}，{age}岁，来自{city}")

# 位置传参（顺序必须对应）
describe_person("小明", 18, "北京")

# 关键字传参（顺序可以不对应）
describe_person(age=18, city="北京", name="小明")

# 混合使用（位置参数必须在关键字参数前面）
describe_person("小明", city="北京", age=18)

② 默认参数

def greet(name, greeting="你好", punctuation="！"):
    print(f"{greeting}，{name}{punctuation}")

greet("小明")                           # 你好，小明！
greet("小红", "早上好")                  # 早上好，小红！
greet("小李", "晚上好", "~")            # 晚上好，小李~
greet("小张", punctuation="^^")        # 你好，小张^^

# 实用例子：打印分隔线
def print_line(char="─", length=40):
    print(char * length)

print_line()             # ────────────────────
print_line("═", 20)     # ════════════════════
print_line("*")          # ****...（40个*）

③ *args —— 可变位置参数

def my_sum(*args):
    print(f"接收到 {len(args)} 个参数：{args}")
    return sum(args)

print(my_sum(1, 2))          # 3
print(my_sum(1, 2, 3, 4))    # 10
print(my_sum(10))            # 10

# 混合位置参数和*args
def print_all(title, *items):
    print(f"=== {title} ===")
    for item in items:
        print(f"  • {item}")

print_all("购物清单", "苹果", "牛奶", "面包", "鸡蛋")

④ **kwargs —— 可变关键字参数

def create_profile(**kwargs):
    print("用户资料：")
    for key, value in kwargs.items():
        print(f"  {key}: {value}")

create_profile(name="周同学", age=28, city="北京", hobby="Python")

# 完整参数顺序：位置 → 默认 → *args → **kwargs
def full_example(a, b, c=10, *args, **kwargs):
    print(f"a={a}, b={b}, c={c}")
    print(f"args={args}, kwargs={kwargs}")

full_example(1, 2, 3, 4, 5, x="hello", y="world")

返回值详解

# 返回单个值
def square(n):
    return n ** 2

# 返回多个值（实际上是元组）
def min_max_avg(numbers):
    return min(numbers), max(numbers), sum(numbers)/len(numbers)

low, high, avg = min_max_avg([3, 1, 4, 1, 5, 9])
print(f"最小={low}, 最大={high}, 平均={avg:.2f}")

# 无 return → 返回 None
def just_print(msg):
    print(msg)
    # 没有return

result = just_print("hello")
print(result)  # None

# 提前 return（条件不满足时）
def divide(a, b):
    if b == 0:
        print("错误：除数不能为0！")
        return None   # 提前退出
    return a / b

print(divide(10, 2))  # 5.0
print(divide(10, 0))  # 错误提示 + None

变量作用域（Scope）

LEGB 规则：Python 按 Local → Enclosed → Global → Built-in 顺序查找变量 
# 局部变量 vs 全局变量
x = "全局变量"

def test_scope():
    x = "局部变量"  # 这是新的局部变量，不影响全局 x
    print(x)       # 局部变量

test_scope()
print(x)  # 全局变量（未被修改）

# global 关键字：在函数内修改全局变量
count = 0

def increment():
    global count
    count += 1

increment()
increment()
print(count)  # 2
⚠️ 最佳实践：尽量避免使用 global，通过参数和返回值传递数据更安全！ 
# ❌ 不推荐：用global修改变量
total = 0
def bad_add(n):
    global total
    total += n

# ✅ 推荐：通过参数和返回值
def good_add(total, n):
    return total + n

total = 0
total = good_add(total, 5)
total = good_add(total, 3)
print(total)  # 8

文档字符串（Docstring）

好的函数应该有清晰的文档！文档字符串是写给其他开发者（和未来的自己）看的。

def calculate_bmi(weight, height):
    """
    计算BMI（体质指数）
    
    BMI = 体重(kg) / 身高(m)²
    
    参数：
        weight (float): 体重，单位千克（kg）
        height (float): 身高，单位米（m）
    
    返回：
        float: BMI 值，保留两位小数
    
    示例：
        >>> calculate_bmi(70, 1.75)
        22.86
        >>> calculate_bmi(50, 1.60)
        19.53
    """
    if height <= 0 or weight <= 0:
        raise ValueError("体重和身高必须为正数")
    bmi = weight / (height ** 2)
    return round(bmi, 2)

# 查看文档
help(calculate_bmi)
print(calculate_bmi.__doc__)  # 也可以直接访问

print(calculate_bmi(70, 1.75))  # 22.86

默认参数陷阱

🚨 经典陷阱：默认参数使用可变对象（如list） 
# ❌ 危险！默认参数是共享的
def add_to_list(item, lst=[]):  # 这个[]只创建一次！
    lst.append(item)
    return lst

print(add_to_list("苹果"))  # ['苹果']
print(add_to_list("香蕉"))  # ['苹果', '香蕉'] ← 意外！
print(add_to_list("橙子"))  # ['苹果', '香蕉', '橙子'] ← 更意外！

# ✅ 正确写法：使用 None 作默认值
def add_to_list_safe(item, lst=None):
    if lst is None:
        lst = []    # 每次调用时创建新列表
    lst.append(item)
    return lst

print(add_to_list_safe("苹果"))  # ['苹果']
print(add_to_list_safe("香蕉"))  # ['香蕉'] ← 正确！

匿名函数 lambda

lambda 是一种简洁的匿名函数，适合简单的单行操作。

语法：lambda 参数: 表达式 
# 普通函数
def double(x):
    return x * 2

# 等价的 lambda
double_lambda = lambda x: x * 2

print(double(5))         # 10
print(double_lambda(5))  # 10

# 多参数 lambda
add = lambda x, y: x + y
print(add(3, 4))  # 7

# lambda 的真正用武之地：排序的 key
students = [
    {"name": "小明", "score": 85},
    {"name": "小红", "score": 92},
    {"name": "小李", "score": 78},
]
students.sort(key=lambda s: s["score"], reverse=True)
print(students[0])  # 小红 92分

# 配合 map() 和 filter()
nums = [1, 2, 3, 4, 5]
doubled = list(map(lambda x: x * 2, nums))
print(doubled)   # [2, 4, 6, 8, 10]

evens = list(filter(lambda x: x % 2 == 0, nums))
print(evens)     # [2, 4]

综合实战

🎯 实战：函数式计算器

def add(a, b): return a + b
def subtract(a, b): return a - b
def multiply(a, b): return a * b
def divide(a, b):
    if b == 0: return "除数不能为0"
    return a / b

operations = {
    "+": add,
    "-": subtract,
    "*": multiply,
    "/": divide,
}

def calculator(expression):
    """
    计算简单的四则运算表达式
    示例：calculator("10 + 5") → 15
    """
    parts = expression.split()
    if len(parts) != 3:
        return "格式错误，请用'数字 运算符 数字'格式"
    a, op, b = parts
    if op not in operations:
        return f"不支持的运算符：{op}"
    try:
        return operations[op](float(a), float(b))
    except ValueError:
        return "请输入有效的数字"

# 测试
for expr in ["10 + 5", "20 - 8", "6 * 7", "15 / 4", "9 / 0"]:
    print(f"{expr} = {calculator(expr)}")
