1️⃣ Flask-WTF表单

pip install flask-wtf flask-sqlalchemy
from flask_wtf import FlaskForm
from wtforms import StringField, PasswordField, validators

app.config['SECRET_KEY'] = 'your-secret-key'

class RegistrationForm(FlaskForm):
    username = StringField('用户名', [
        validators.DataRequired(message='请输入用户名'),
        validators.Length(min=3, max=20)
    ])
    email = StringField('邮箱', [
        validators.DataRequired(), validators.Email()
    ])
    password = PasswordField('密码', [
        validators.DataRequired(), validators.Length(min=6)
    ])
    password2 = PasswordField('确认密码', [
        validators.EqualTo('password', message='密码不一致')
    ])

@app.route('/register', methods=['GET','POST'])
def register():
    form = RegistrationForm()
    if form.validate_on_submit():
        flash(f'注册成功！欢迎{form.username.data}', 'success')
        return redirect(url_for('index'))
    return render_template('register.html', form=form)

# 自定义验证器
def validate_phone(self, field):
    import re
    if not re.match(r'^1[3-9]\d{9}$', field.data):
        raise ValidationError('手机号格式错误')

3️⃣ 文件上传

from flask_wtf.file import FileField, FileAllowed, FileRequired
from werkzeug.utils import secure_filename
import os

class UploadForm(FlaskForm):
    file = FileField('上传', [
        FileRequired(), FileAllowed(['jpg','png','pdf'], '只允许图片和PDF')
    ])

@app.route('/upload', methods=['GET','POST'])
def upload():
    form = UploadForm()
    if form.validate_on_submit():
        f = form.file.data
        filename = secure_filename(f.filename)
        f.save(os.path.join('uploads', filename))
        flash('上传成功！', 'success')
    return render_template('upload.html', form=form)

4️⃣ SQLAlchemy ORM

from flask_sqlalchemy import SQLAlchemy
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///site.db'
db = SQLAlchemy(app)

class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(20), unique=True, nullable=False)
    email = db.Column(db.String(120), unique=True, nullable=False)
    posts = db.relationship('Post', backref='author', lazy=True)

class Post(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(100), nullable=False)
    content = db.Column(db.Text, nullable=False)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'))

with app.app_context():
    db.create_all()

6️⃣ 实战：待办事项应用

class Todo(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(100), nullable=False)
    complete = db.Column(db.Boolean, default=False)

@app.route('/')
def index():
    todos = Todo.query.order_by(Todo.id.desc()).all()
    return render_template('todo.html', todos=todos)

@app.route('/add', methods=['POST'])
def add():
    title = request.form.get('title')
    if title:
        db.session.add(Todo(title=title))
        db.session.commit()
    return redirect(url_for('index'))

@app.route('/complete/')
def complete(tid):
    todo = Todo.query.get_or_404(tid)
    todo.complete = not todo.complete
    db.session.commit()
    return redirect(url_for('index'))

@app.route('/delete/')
def delete(tid):
    todo = Todo.query.get_or_404(tid)
    db.session.delete(todo)
    db.session.commit()
    return redirect(url_for('index'))
