        return user.to_dict()

5️⃣ 认证与授权

from functools import wraps
from flask import request, jsonify

# API Key认证
def require_api_key(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        api_key = request.headers.get('X-API-Key')
        if api_key != 'your-secret-key':
            return jsonify({'error': 'Unauthorized'}), 401
        return f(*args, **kwargs)
    return decorated

# JWT认证
import jwt

def require_jwt(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get('Authorization', '').replace('Bearer ', '')
        try:
            payload = jwt.decode(token, 'secret', algorithms=['HS256'])
            request.user_id = payload['user_id']
        except jwt.InvalidTokenError:
            return jsonify({'error': 'Invalid token'}), 401
        return f(*args, **kwargs)
    return decorated

class ProtectedResource(Resource):
    @require_jwt
    def get(self):
        return {'user_id': request.user_id, 'message': 'Secret data'}

6️⃣ 分页与过滤

class UserListAPI(Resource):
    def get(self):
        # 分页参数
        page = request.args.get('page', 1, type=int)
        per_page = request.args.get('per_page', 10, type=int)
        
        # 过滤参数
        name = request.args.get('name')
        status = request.args.get('status')
        
        query = UserModel.query
        
        if name:
            query = query.filter(UserModel.name.contains(name))
        if status:
            query = query.filter_by(status=status)
        
        # 分页
        pagination = query.paginate(page=page, per_page=per_page)
        
        return {
            'items': [u.to_dict() for u in pagination.items],
            'total': pagination.total,
            'page': page,
            'per_page': per_page,
            'pages': pagination.pages
        }

9️⃣ 实战项目：任务API

# 使用Flask-RESTful构建任务管理API
from flask import Flask
from flask_restful import Resource, Api, reqparse, fields, marshal_with
from flask_sqlalchemy import SQLAlchemy
from datetime import datetime

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///tasks.db'
db = SQLAlchemy(app)
api = Api(app)

# 模型
class Task(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(100), nullable=False)
    description = db.Column(db.Text)
    done = db.Column(db.Boolean, default=False)
    created_at = db.Column(db.DateTime, default=datetime.utcnow)

# 序列化
task_fields = {
    'id': fields.Integer,
    'title': fields.String,
    'description': fields.String,
    'done': fields.Boolean,
    'created_at': fields.DateTime(dt_format='iso8601')
}

parser = reqparse.RequestParser()
parser.add_argument('title', type=str, required=True)
parser.add_argument('description', type=str)
parser.add_argument('done', type=bool)

class TaskListAPI(Resource):
    @marshal_with(task_fields)
    def get(self):
        tasks = Task.query.all()
        return tasks
    
    @marshal_with(task_fields)
    def post(self):
        args = parser.parse_args()
        task = Task(**args)
        db.session.add(task)
        db.session.commit()
        return task, 201

class TaskAPI(Resource):
    @marshal_with(task_fields)
    def get(self, task_id):
        task = Task.query.get_or_404(task_id)
        return task
    
    @marshal_with(task_fields)
    def put(self, task_id):
        task = Task.query.get_or_404(task_id)
        args = parser.parse_args()
        for key, value in args.items():
            if value is not None:
                setattr(task, key, value)
        db.session.commit()
        return task
    
    def delete(self, task_id):
        task = Task.query.get_or_404(task_id)
        db.session.delete(task)
        db.session.commit()
        return '', 204

api.add_resource(TaskListAPI, '/api/tasks')
api.add_resource(TaskAPI, '/api/tasks/')

if __name__ == '__main__':
    with app.app_context():
        db.create_all()
    app.run(debug=True)
