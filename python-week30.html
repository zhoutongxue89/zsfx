6️⃣ 数据清洗

# 缺失值
df.isnull().sum()     # 缺失数
df.dropna()           # 删除缺失行
df.fillna(0)          # 填充0
df.fillna({'age':20,'score':df['score'].mean()})

# 重复值
df.drop_duplicates()

# 新增列
df['pass'] = df['score'] >= 60
df['grade'] = df['score'].apply(
    lambda x: 'A' if x>=90 else 'B' if x>=80 else 'C')

7️⃣ 实战：成绩分析

df = pd.DataFrame({
    'name': ['张三','李四','王五','赵六','钱七'],
    'class': ['A','A','B','B','C'],
    'chinese': [85,78,92,65,88],
    'math': [90,85,78,72,95],
    'english': [78,92,85,80,73],
})
df['total'] = df['chinese']+df['math']+df['english']
df['avg'] = df['total']/3

# 按班级分组统计
print(df.groupby('class').agg({
    'chinese':['mean','max'],'math':'mean','total':'mean'
}))

# 排名
df['rank'] = df['total'].rank(ascending=False,method='dense').astype(int)
print(df.sort_values('rank'))
